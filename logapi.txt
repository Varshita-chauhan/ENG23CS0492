 import requests
from typing import Literal

StackType = Literal["backend", "frontend"]
LevelType = Literal["debug", "info", "warn", "error", "fatal"]
PackageType = Literal["api", "component", "hook", "page", "state", "style", "auth", "config", "middleware", "utils"]

class RemoteLogger:
    # 1. Add an auth_token parameter to initialization
    def __init__(self, endpoint_url: str = "http://4.224.186.213/evaluation-service/logs", auth_token: str = None):
        self.url = endpoint_url
        self.auth_token = auth_token

    def log(self, stack: StackType, level: LevelType, package: PackageType, message: str) -> dict:
        """
        Sends a log payload to the remote evaluation service with Authorization headers.
        """
        payload = {
            "stack": stack,
            "level": level,
            "package": package,
            "message": message
        }

        # 2. Construct headers if a token is provided
        headers = {}
        if self.auth_token:
            # Note: If the API expects a standard Bearer token, use: f"Bearer {self.auth_token}"
            # If it just expects the raw API key, use: self.auth_token
            headers["Authorization"] = f"Bearer {self.auth_token}" 

        try:
            # 3. Pass the headers into the post request
            response = requests.post(self.url, json=payload, headers=headers, timeout=5)
            response.raise_for_status() 
            return response.json()
        except requests.exceptions.RequestException as e:
            print(f"[Logger Error] Failed to send log: {e}")
            return {"error": str(e)}
